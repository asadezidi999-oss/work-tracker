function liveTime(){ let d=getDay(); let today=new Date().toISOString().split('T')[0]; // ❗ مهم: فقط اليوم الحالي if(d !== today) return 0; let total=0; if(!data[d]) return 0; Object.values(data[d]).forEach(l=>{ let s=null,p=0,ps=null; l.forEach(x=>{ let tm=parseHM(x.time); if(x.type==='Start') s=tm; if(x.type==='Pause Start') ps=tm; if(x.type==='Pause Ende' && ps!=null){ let end=tm; if(end < ps) end += 1440; p += end - ps; ps=null; } }); if(s!=null){ let now=parseHM(nowHM()); if(now < s) now += 1440; total += Math.max(0, now - s - p); } }); return total; }
